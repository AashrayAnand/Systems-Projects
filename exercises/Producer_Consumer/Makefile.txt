CXX = g++
OBJS = SimpleQueue.o ex16.o
EXE  = ex16
HEADERS = SimpleQueue.h
CPPFLAGS = -Wall -g -std=c++11
LDFLAGS += -pthread

# default target
$(EXE): $(OBJS)
	$(CXX) $(CPPFLAGS) -o $(EXE) $(OBJS) $(LDFLAGS)
# source files
%.o: %.cc $(HEADERS)
	$(CXX) $(CPPFLAGS) -c $<
# phony target - remove generated files and backups
clean:
	rm -rf $(EXE) *.o *~ *.dSYM
